  CLI 使用示例

  # 分析项目
  pyimport2pkg analyze ./my_project

  # 输出到文件
  pyimport2pkg analyze ./my_project -o requirements.txt

  # JSON 格式
  pyimport2pkg analyze ./my_project -f json

  # 查询模块映射
  pyimport2pkg query cv2

  # 构建数据库
  pyimport2pkg build-db --max-packages 5000
